#Generate assembly file <file_name>.s from <file_name>.c
    - riscv64-unknown-elf-gcc -S <file_name>.c


# Generate ELF executable <file_name> from <file_name>.s
    - riscv64-unknown-elf-gcc -Wl,-Ttext=0x0 -nostdlib -march=rv64i -mabi=lp64 -o <file_name> <file_name>.s


# Generate binary file <file_name>.bin from ELF file <file_name> (what will be read by emulator)
    - riscv64-unknown-elf-objcopy -O binary <file_name> <file_name>.bin
